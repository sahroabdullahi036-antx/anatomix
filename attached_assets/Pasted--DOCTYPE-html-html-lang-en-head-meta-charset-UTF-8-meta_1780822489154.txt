  <script>
    const correctBtn = document.getElementById('correctBtn');
    const wrongBtn = document.getElementById('wrongBtn');

    // --- Trigger Wrong Buzz & Red X ---
    wrongBtn.addEventListener('click', () => {
      // Prevent button spamming overlapping animations
      if (wrongBtn.classList.contains('wrong-shake')) return;

      wrongBtn.classList.add('wrong-shake');

      // Remove classes after animation timeline finishes so it can re-trigger
      setTimeout(() => {
        wrongBtn.classList.remove('wrong-shake');
      }, 500); 
    });

    // --- Trigger Targeted Confetti Burst ---
    correctBtn.addEventListener('click', () => {
      correctBtn.classList.add('btn-correct');
      createConfettiBurst(correctBtn);

      setTimeout(() => {
        correctBtn.classList.remove('btn-correct');
      }, 600);
    });

    function createConfettiBurst(parentElement) {
      const colors = ['#38bdf8', '#4ade80', '#facc15', '#f472b6', '#a78bfa'];
      const particleCount = 15;

      for (let i = 0; i < particleCount; i++) {
        const particle = document.createElement('div');
        particle.classList.add('confetti-particle');
        
        // Randomize colors and shapes
        particle.style.backgroundColor = colors[Math.floor(Math.random() * colors.length)];
        if (Math.random() > 0.5) particle.style.borderRadius = '50%';

        // Start particle in the exact middle of the button
        parentElement.appendChild(particle);

        // Generate physics trajectories
        const angle = Math.random() * Math.PI * 2; // Random 360 degree direction
        const velocity = 2 + Math.random() * 4;   // Random launch speed
        let posX = parentElement.offsetWidth / 2;
        let posY = parentElement.offsetHeight / 2;
        let moveX = Math.cos(angle) * velocity;
        let moveY = Math.sin(angle) * velocity;
        let opacity = 1;

        // Custom micro-engine loop for each particle animation frame
        function updateParticle() {
          posX += moveX;
          posY += moveY;
          moveY += 0.15; // Gravity pull factor
          opacity -= 0.02; // Fade factor

          particle.style.transform = `translate(${posX}px, ${posY}px)`;
          particle.style.opacity = opacity;

          if (opacity > 0) {
            requestAnimationFrame(updateParticle);
          } else {
            particle.remove();
          }
        }
        
        requestAnimationFrame(updateParticle);
      }
    }
  </script>
